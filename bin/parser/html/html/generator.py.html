#!/usr/bin/python3.6

from schema import root
import yaml
from collections import OrderedDict
from typing import Any, Dict, List
import typing

def generateTileTemplate() -> None:
    with open("tile.yml.template.input", "r") as inFile:
        tileManifestData = yaml.load(inFile)
    propertyListOutput = {}
    formOutput = []
    root.generateTileTemplate(".properties", propertyListOutput, tileManifestData["forms"])

    outputDict = {}
    for propertyName, parameters in propertyListOutput.items():
        assert parameters
        output = "(( " + parameters[0]
        for i in range(1, len(parameters)):
            output = output + " || " + parameters[i]
        output = output + " ))"
        outputDict[propertyName] = output

    for package in tileManifestData["packages"]:
        if package["name"] == "solace-vmr":
            for job in package["jobs"]:
                job["properties"].update(outputDict)

    with open("tile.yml.template.output", "w") as outFile:
        # width has to be large or the yaml text rows will wrap, which is harder to read
        yaml.dump(tileManifestData, outFile, default_flow_style=False, width=100000000)

generateTileTemplate()
